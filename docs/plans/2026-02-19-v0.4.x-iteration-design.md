```typescript
// src/hooks/useTheme.ts
import { useState, useEffect, useCallback } from 'react';

export type Theme = 'light' | 'dark' | 'dracula' | 'nord' | 'monokai' | 'glass';

const THEME_KEY = 'plan-todos-theme';

export function useTheme() {
  const [theme, setThemeState] = useState<Theme>('light');

  // åˆå§‹åŒ–æ—¶ä» localStorage è¯»å–
  useEffect(() => {
    const stored = localStorage.getItem(THEME_KEY) as Theme;
    if (stored && isValidTheme(stored)) {
      setTheme(stored);
    } else {
      // æ£€æµ‹ç³»ç»Ÿåå¥½
      const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
      setTheme(prefersDark ? 'dark' : 'light');
    }
  }, []);

  const setTheme = useCallback((newTheme: Theme) => {
    setThemeState(newTheme);
    document.documentElement.setAttribute('data-theme', newTheme);
    localStorage.setItem(THEME_KEY, newTheme);
  }, []);

  return { theme, setTheme };
}

function isValidTheme(t: string): t is Theme {
  return ['light', 'dark', 'dracula', 'nord', 'monokai', 'glass'].includes(t);
}
```
